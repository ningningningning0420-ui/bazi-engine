## 红线（每任务都守 · 自检：把核换 roll()，行为有无实质变化？无=退化）
1. **favorSign 唯一吉凶源**：净吉凶/方向/aptitude/timing/successDelta 最终全落 `Σ(Δfield×favorSign)` 内积；关系算子只改场扰动+吐谓词，由 favorSign 定吉凶；**绝不另写"算子→吉凶"或"事件→分数"或"干支→吉凶"表**。
2. **100% 确定性**：无 RNG/`Date`/全局态（`atDate` 显式入参·plain `{year,month,day}` 非 Date 对象）；掷骰外置；所有并列有固定 tie-break；round 只在输出。同 `(chart,atDate,grain)` 两次深相等。
3. **命理零泄漏（双寄存器）**：`隐藏暗示`/域倾向**台前**=命理-free 行为语言（无五行/十神/旺衰/流年词·禁词单测）；`域倾向.依据`**后台**=命理串（合法）。
4. **八字不自翻向**：`directionBias` 仅 hint·与 successDelta 同源 net 符号一致·绝不参与翻向（翻向归宿主 gate）。
5. **不掷骰**：核只出 `EventCoeff`；`rollWithCoeff(coeff, prng)` standalone 兜底·prng 外置注入·宿主消费方只读 EventCoeff。
6. **地基不是牢笼**：L3 是压力非剧本·可抵抗·不锁人格。

```ts
import { Solar } from 'lunar-typescript';
import { buildPillar } from '../build-pillar';
import type { Chart, AtDate, GrainSet, LuckPillars, Pillar, Gan, Zhi } from '../types';

const SECT = 1;
const splitGZ = (gz: string): [Gan, Zhi] => [gz[0] as Gan, gz[1] as Zhi];

/** 大运/流年/流月/流日（★唯一碰 lunar 处·历法 oracle）。流年/月/日=排 atDate 自身八字(立春/节气自动)；大运=getYun。 */
export function selectLuckPillars(chart: Chart, atDate: AtDate, grain: GrainSet): LuckPillars {
  const dm = chart.日主.gan;
  const caveats: string[] = [];

  // 流年/流月/流日 = 排 atDate 自己的八字（立春换年 / 节气定月 自动）
  const atEc = Solar.fromYmdHms(atDate.year, atDate.month, atDate.day, 12, 0, 0).getLunar().getEightChar();
  const [ny, nz] = splitGZ(atEc.getYear());
  const 流年 = buildPillar(dm, ny, nz);
  let 流月: Pillar | null = null, 流日: Pillar | null = null;
  if (grain.includes('流月')) { const [g, z] = splitGZ(atEc.getMonth()); 流月 = buildPillar(dm, g, z); }
  if (grain.includes('流日')) { const [g, z] = splitGZ(atEc.getDay()); 流日 = buildPillar(dm, g, z); }

  // 大运：从 birthInput 重排盘 → getYun（乾→1男 / 坤→0女·hourUnknown→12·sect=1），按 atDate 公历年定位
  const bi = chart.birthInput;
  const bh = bi.hourUnknown ? 12 : bi.hour!;
  const birthEc = Solar.fromYmdHms(bi.year, bi.month, bi.day, bh, 0, 0).getLunar().getEightChar();
  const yun = birthEc.getYun(bi.gender === '乾' ? 1 : 0, SECT);
  const dys = yun.getDaYun(12);
  const dy = dys.find((d) => d.getGanZhi() !== '' && atDate.year >= d.getStartYear() && atDate.year <= d.getEndYear());
  let 大运: Pillar | null = null;
  if (dy) { const [g, z] = splitGZ(dy.getGanZhi()); 大运 = buildPillar(dm, g, z); }
  else caveats.push('未起运·童限（无大运底色，L3 信号弱化）');

  if (bi.hourUnknown) caveats.push('时辰未知：时柱/宫位相关断语降级');
  return { 大运, 流年, 流月, 流日, caveats };
}
```

```ts
import { DIZHI_LIUHE, DIZHI_SANHE, DIZHI_SANHUI, DIZHI_LIUCHONG, DIZHI_LIUHAI, DIZHI_SANXING_TRIPLE, DIZHI_XING_PAIR } from '../facts/relations';
import { SLOT_ORDER, type FourPillars, type LuckPillars, type Grain, type Pillar } from '../types';
import type { Zhi, WuXing } from '../constants/gan-zhi';

export type CrossKind = '六合' | '三合' | '三会' | '六冲' | '六害' | '三刑';
export interface CrossRelation { 类型: CrossKind; 流运: Grain; 流运支: Zhi; 化五行: WuXing | null; }

const GRAIN_KEYS: [Grain, keyof LuckPillars][] = [['大运', '大运'], ['流年', '流年'], ['流月', '流月'], ['流日', '流日']];

export function crossRelations(four: FourPillars, luck: LuckPillars): CrossRelation[] {
  const natalZhi = SLOT_ORDER.filter((s) => four[s] !== null).map((s) => (four[s] as Pillar).zhi);
  const out: CrossRelation[] = [];
  for (const [grain, key] of GRAIN_KEYS) {
    const p = luck[key] as Pillar | null;
    if (!p) continue;
    const lz = p.zhi;
    // 六合
    for (const [a, b, hua] of DIZHI_LIUHE) {
      if ((lz === a && natalZhi.includes(b)) || (lz === b && natalZhi.includes(a)))
        out.push({ 类型: '六合', 流运: grain, 流运支: lz, 化五行: hua });
    }
    // 六冲
    for (const [a, b] of DIZHI_LIUCHONG) {
      if ((lz === a && natalZhi.includes(b)) || (lz === b && natalZhi.includes(a)))
        out.push({ 类型: '六冲', 流运: grain, 流运支: lz, 化五行: null });
    }
    // 六害
    for (const [a, b] of DIZHI_LIUHAI) {
      if ((lz === a && natalZhi.includes(b)) || (lz === b && natalZhi.includes(a)))
        out.push({ 类型: '六害', 流运: grain, 流运支: lz, 化五行: null });
    }
    // 三刑(三字·流运补第3)
    for (const tri of DIZHI_SANXING_TRIPLE) {
      if (tri.includes(lz) && tri.filter((z) => z !== lz).every((z) => natalZhi.includes(z)))
        out.push({ 类型: '三刑', 流运: grain, 流运支: lz, 化五行: null });
    }
    for (const [a, b] of DIZHI_XING_PAIR) {
      if ((lz === a && natalZhi.includes(b)) || (lz === b && natalZhi.includes(a)))
        out.push({ 类型: '三刑', 流运: grain, 流运支: lz, 化五行: null });
    }
    // 三合/三会(原局已有2支 + 流运补第3)
    for (const [k, table] of [['三合', DIZHI_SANHE], ['三会', DIZHI_SANHUI]] as const) {
      for (const row of table) {
        const trip = [row[0], row[1], row[2]] as Zhi[];
        if (trip.includes(lz) && trip.filter((z) => z !== lz).every((z) => natalZhi.includes(z)))
          out.push({ 类型: k, 流运: grain, 流运支: lz, 化五行: row[3] });
      }
    }
  }
```

### Task 4: `perturbField` — 力量场 + 算子 reducer + 净吉凶（★承重墙）

**Files:** Create `src/forecast/field.ts`；Test `test/field.test.ts`

**精确语义（审核焊死）：**
- 模块私有旋钮（文件顶部·命名 export）：`GRAIN_WEIGHT={大运:0.5,流年:1.0,流月:0.5,流日:0.3}`、`HUA_TRANSFER=0.3`（合化转移率）、`CHONG_LOSS=0.4`（冲损）、`INTENSITY_MID=0.5`/`INTENSITY_HIGH=1.5`（强度档）。
- **base 注入**：每在场流运柱按 `GRAIN_WEIGHT` 注入：天干 `+STEM_WEIGHT×w` 到天干五行；藏干 `+HIDDEN_WEIGHT[role]×w` 到藏干五行。（force 是**扰动**·不含原局基线。）
- **算子 reducer（贪合忘冲·确定性）**：对每个**在场流运支**：
  1. 查 crossRelations 该支的关系。**合（六合/三合/三会）优先于冲**（贪合忘冲）。
  2. 有合：**化成判据 = 化五行 !== null 且 化五行 === 原局月令本气五行（硬规则）** → 把该流运支注入的 base 力量 `×HUA_TRANSFER` 从该支本气五行**转移**到化五行（`Δfield[本气] -= t; Δfield[化] += t`）；作用 `合化`。否则（合绊/午未null/化神不当令）→ 不转移、该支视为**已绊**（不再冲）；作用 `合化`(绊)。
  3. 无合但有冲：`Δfield[该流运支本气五行] ×= (1−CHONG_LOSS)`；作用 `刑冲`·标记进 rawFlags 不重复（见下）；动荡信号入域。
  4. 害/三刑：v1 仅记作用 `刑冲`（不改 Δfield·力量留 v2）。
  - **同优先级多候选**：crossRelations 已按 GRAIN_KEYS→table 顺序稳定产出；reducer 遍历 `WU_XING`/`SLOT_ORDER` 固定序，平手取首。
- **净吉凶 = Σ(Δfield[w]×favorSign[w])**（favorSign=chart.分析.用神.favorSign）。
- **作用[].强度**：力量幅度 `<INTENSITY_MID`微 / `<INTENSITY_HIGH`中 / `≥`显。`引动用神/忌神` = 算子作用五行 favorSign>0/<0。
- **rawFlags（基准集=原局四柱+大运·查流年）**：`反吟`=流年干**克**原局某柱干（GAN_WUXING ke）且流年支**冲**该柱支；`伏吟`=流年干支==原局某柱干支；`岁运并临`=流年干支==大运干支。**两通道**：rawFlags 只供 L4 amp 缩放烈度·**不改 Δfield**；流年支的冲已在算子层计冲损（力量通道），反吟 flag 只放大烈度（severity 通道）·不二次扣力。
- round2 只在输出 净吉凶 / Δfield。

```ts
import { GAN_WUXING, HIDDEN_STEMS, ZHI_WUXING, WU_XING, ke, type WuXing } from '../constants/gan-zhi';
import { STEM_WEIGHT, HIDDEN_WEIGHT } from '../constants/weights';
import { SLOT_ORDER, type ChartFacts, type LuckPillars, type FieldResult, type FieldAction, type Pillar, type Grain, type Intensity } from '../types';
import { crossRelations } from './field-relations';

export const GRAIN_WEIGHT: Record<Grain, number> = { 大运: 0.5, 流年: 1.0, 流月: 0.5, 流日: 0.3 };
export const HUA_TRANSFER = 0.3;
export const CHONG_LOSS = 0.4;
export const INTENSITY_MID = 0.5;
export const INTENSITY_HIGH = 1.5;
const round2 = (x: number): number => Math.round(x * 100) / 100;
const GRAIN_KEYS: [Grain, keyof LuckPillars][] = [['大运', '大运'], ['流年', '流年'], ['流月', '流月'], ['流日', '流日']];
const intensityOf = (mag: number): Intensity => (mag < INTENSITY_MID ? '微' : mag < INTENSITY_HIGH ? '中' : '显');

export function perturbField(facts: ChartFacts, luck: LuckPillars): FieldResult {
  const favorSign = facts.分析.用神.favorSign;
  const monthBenQi = ZHI_WUXING[facts.四柱.月.zhi]; // 原局月令本气五行(化神当令判据)
  const Δfield: Record<WuXing, number> = { 木: 0, 火: 0, 土: 0, 金: 0, 水: 0 };
  const 作用: FieldAction[] = [];

  // base 注入（记录每流运支注入的本气力量，供合化转移）
  const benQiInject: { grain: Grain; zhi: string; w: WuXing; force: number }[] = [];
  for (const [grain, key] of GRAIN_KEYS) {
    const p = luck[key] as Pillar | null;
    if (!p) continue;
    const w = GRAIN_WEIGHT[grain];
    Δfield[GAN_WUXING[p.gan]] += STEM_WEIGHT * w;
    const hs = HIDDEN_STEMS[p.zhi];
    for (const h of hs) Δfield[GAN_WUXING[h.gan]] += HIDDEN_WEIGHT[h.role] * w;
    benQiInject.push({ grain, zhi: p.zhi, w: ZHI_WUXING[p.zhi], force: HIDDEN_WEIGHT['本气'] * w });
  }

  // 算子 reducer（贪合忘冲）
  const rels = crossRelations(facts.四柱, luck);
  const consumed = new Set<string>(); // 已绊流运支(grain+zhi)
  for (const inj of benQiInject) {
    const key = inj.grain + inj.zhi;
    const myRels = rels.filter((r) => r.流运 === inj.grain && r.流运支 === inj.zhi);
    const he = myRels.find((r) => r.类型 === '六合' || r.类型 === '三合' || r.类型 === '三会');
    if (he) {
      consumed.add(key);
      if (he.化五行 !== null && he.化五行 === monthBenQi) {
        const t = inj.force * HUA_TRANSFER;
        Δfield[inj.w] -= t; Δfield[he.化五行] += t;
        作用.push({ 类型: '合化', 对象: `${inj.zhi}化${he.化五行}`, 强度: intensityOf(t) });
      } else {
        作用.push({ 类型: '合化', 对象: `${inj.zhi}合绊`, 强度: '微' });
      }
      continue;
    }
    const chong = myRels.find((r) => r.类型 === '六冲');
    if (chong) {
      const loss = Δfield[inj.w] * CHONG_LOSS;
      Δfield[inj.w] -= loss;
      作用.push({ 类型: '刑冲', 对象: `${inj.zhi}冲`, 强度: intensityOf(loss) });
    }
    const xinghai = myRels.find((r) => r.类型 === '六害' || r.类型 === '三刑');
    if (xinghai) 作用.push({ 类型: '刑冲', 对象: `${inj.zhi}${xinghai.类型}`, 强度: '微' });
  }

  // 生扶/克泄/引动用神忌神（按 Δfield 符号 + favorSign 标注·读最强扰动行）
  for (const w of WU_XING) {
    if (Δfield[w] > INTENSITY_MID) {
      const k = favorSign[w] > 0 ? '引动用神' : favorSign[w] < 0 ? '引动忌神' : '生扶';
      作用.push({ 类型: k, 对象: w, 强度: intensityOf(Δfield[w]) });
    }
  }

  const 净吉凶 = round2(WU_XING.reduce((a, w) => a + Δfield[w] * favorSign[w], 0));
  for (const w of WU_XING) Δfield[w] = round2(Δfield[w]);

  // rawFlags（基准集=原局四柱+大运·查流年）
  const rawFlags = { 反吟: false, 伏吟: false, 岁运并临: false };
  const ly = luck.流年;
  const basis: Pillar[] = SLOT_ORDER.filter((s) => facts.四柱[s] !== null).map((s) => facts.四柱[s] as Pillar);
  if (luck.大运) basis.push(luck.大运);
  for (const p of basis) {
    if (p.gan === ly.gan && p.zhi === ly.zhi) rawFlags.伏吟 = true;
    if (ke(GAN_WUXING[ly.gan]) === GAN_WUXING[p.gan] && isChong(ly.zhi, p.zhi)) rawFlags.反吟 = true;
  }
  if (luck.大运 && luck.大运.gan === ly.gan && luck.大运.zhi === ly.zhi) rawFlags.岁运并临 = true;

  return { Δfield, 净吉凶, 作用, rawFlags };
}

function isChong(a: string, b: string): boolean {
  const pairs = [['子','午'],['丑','未'],['寅','申'],['卯','酉'],['辰','戌'],['巳','亥']];
  return pairs.some(([x, y]) => (a === x && b === y) || (a === y && b === x));
}
```

```ts
import { sheng, ke, shengMe, keMe, type WuXing } from '../constants/gan-zhi';
import { selectLuckPillars } from './luck-pillars';
import { perturbField } from './field';
import { computeChartFacts } from '../analyze-chart';
import { SLOT_ORDER, type Chart, type AtDate, type GrainSet, type Tendency, type DomainTendency, type DomainKind, type Direction, type Intensity, type FieldResult, type Pentad } from '../types';

const DIR_EPS = 0.15;
const intensityOf = (m: number): Intensity => (m < 0.5 ? '微' : m < 1.5 ? '中' : '显');
const DOMAIN_ORDER: DomainKind[] = ['事业权位', '财', '关系情感', '健康', '心性波动', '名声口舌'];

// 域 → 关注五行（相对日主五行 dmW）
function domainFocus(域: DomainKind, dmW: WuXing, gender: '乾' | '坤'): WuXing[] {
  switch (域) {
    case '事业权位': return [keMe(dmW), shengMe(dmW)];
    case '财': return [ke(dmW), sheng(dmW)];
    case '关系情感': return [gender === '乾' ? ke(dmW) : keMe(dmW)];
    case '健康': return [dmW, shengMe(dmW)];
    case '心性波动': return [sheng(dmW), shengMe(dmW)];
    case '名声口舌': return [keMe(dmW), shengMe(dmW)];
  }
}

// 命理-free 行为短语池（域日常词 × 方向 × 强度）——纯渲染层·真值是结构化域倾向
const DOMAIN_WORD: Record<DomainKind, string> = { 事业权位: '事业', 财: '财务', 关系情感: '人际', 健康: '状态', 心性波动: '心绪', 名声口舌: '风评' };
function phrase(d: DomainTendency): string {
  const w = DOMAIN_WORD[d.域];
  const deg = d.强度 === '显' ? '明显' : d.强度 === '中' ? '较' : '略';
  if (d.方向 === '↑') return `近期在${w}上${deg}有起色`;
  if (d.方向 === '↓') return `近期在${w}上${deg}承压`;
  if (d.方向 === '动荡') return `近期${w}起伏不定`;
  return `近期${w}大致平稳`;
}

export function deriveTendency(chart: Chart, atDate: AtDate, grain: GrainSet): Tendency {
  const facts = computeChartFacts(chart.birthInput);
  const dmW = chart.日主.wuXing;
  const favorSign = facts.分析.用神.favorSign;
  const luck = selectLuckPillars(chart, atDate, grain);
  const field: FieldResult = perturbField(facts, luck);

  const 大运吉凶倾向 = luck.大运 == null ? '未起运'
    : favorSign[luck.大运.zhiWuXing] > 0 ? '用神运' : favorSign[luck.大运.zhiWuXing] < 0 ? '忌神运' : '中性';
  // 注：大运五行取干五行更准——用大运天干五行：
  const 大运W = luck.大运 ? chart.日主.wuXing : null; // 占位，见实现说明

  const dayZhi = chart.四柱.日.zhi;
  const 动荡日支冲 = field.作用.some((a) => a.类型 === '刑冲');
  const 域倾向: DomainTendency[] = DOMAIN_ORDER.map((域) => {
    const focus = domainFocus(域, dmW, chart.birthInput.gender);
    const net = focus.reduce((a, w) => a + field.Δfield[w] * favorSign[w], 0);
    const 动荡 = (域 === '关系情感' || 域 === '健康' || 域 === '心性波动' || 域 === '名声口舌') && 动荡日支冲;
    const 方向: Direction = 动荡 ? '动荡' : net > DIR_EPS ? '↑' : net < -DIR_EPS ? '↓' : '平';
    return { 域, 方向, 强度: intensityOf(Math.abs(net)), net: Math.round(net * 100) / 100,
      依据: `${域}关注五行[${focus.join('')}]·net=${Math.round(net*100)/100}(Δfield×favorSign)` };

  return {
    当运干支: { 大运: luck.大运 ? luck.大运.gan + luck.大运.zhi : null, 流年: luck.流年.gan + luck.流年.zhi,
      流月: luck.流月 ? luck.流月.gan + luck.流月.zhi : null, 流日: luck.流日 ? luck.流日.gan + luck.流日.zhi : null },
    大运吉凶倾向, 作用: field.作用, 域倾向, Δfield_blend: field.Δfield, 净吉凶: field.净吉凶,
    隐藏暗示, rawFlags: field.rawFlags, caveats: luck.caveats,
  };
}
```

**精确语义（★承重墙·share 口径 + favorSign 本值）：**
- 模块旋钮（文件顶部）：`DELTA_SCALE=0.25`、`AMP={反吟:1.5,伏吟:1.5,岁运并临:1.8}`、`AMP_CLAMP=2.0`、`DIR_EPS=0.05`。
- 恒用最细 grain `['大运','流年','流月','流日']`（L4 数值 grain 无关）。
- focus = `EVENT_FOCUS_TABLE[eventKind].关注十神`（类别）→ 解析五行 = 各类别相对日主的五行；婚恋按 gender 取 配偶星。
- **aptitude（命局禀赋·-1..1·share 口径）**：`share[w]=旺衰.五行得分[w]/Σ五行得分`；`aptitude=clamp(Σ_{w∈focus}(share[w]×favorSign[w]) / (2×|focus|), -1, 1)`（favorSign∈-2..2·share∈[0,1]）。Σ五行得分=0→0。
- **timing（当期引动·-1..1）**：`D=Σ_{all w}|Δfield[w]×favorSign[w]|`；`timing=D===0?0:clamp(Σ_{w∈focus}(Δfield[w]×favorSign[w]) / D, -1, 1)`。
- **net = clamp(0.4×aptitude + 0.6×timing, -1, 1)**。
- **amp**：rawFlags 命中取对应 `AMP` 值**取 max**（非连乘）·clamp≤`AMP_CLAMP`·无命中=1.0。`主导amp`=取 max 的 flag 名（无→null）。
- 输出：`successDelta=round2(net×DELTA_SCALE)`；`severityScale=round2(amp)`；`directionBias=net>DIR_EPS?'↑':net<-DIR_EPS?'↓':'中'`（**与 successDelta 同源 net·符号一致·仅 hint·不翻向**）；`rationale={focus,aptitude:round2,timing:round2,amp:round2,主导amp}`。

```ts
import type { EventCoeff, RollResult } from '../types';
const BASE_RATE = 0.5;
const clamp = (x: number, lo: number, hi: number): number => Math.max(lo, Math.min(hi, x));
/**
 * standalone 兜底掷骰：把 EventCoeff.successDelta 修正 baseRate 当阈值掷骰。
 * 契约：prng 外置注入(核不内置 RNG)；通用卡兜底用；★大奥等有裁决引擎的消费方禁止 import 本函数，只读 EventCoeff 喂自己的裁决/真蝴蝶 gate。
 */
export function rollWithCoeff(coeff: EventCoeff, prng: () => number): RollResult {
  const threshold = clamp(BASE_RATE + coeff.successDelta, 0.05, 0.95);
  const roll = prng();
  return { success: roll < threshold, roll, threshold };
}
```
